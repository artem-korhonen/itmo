Все было холодным и мокрым.

@startuml
entity Human {
  human_id : SERIAL
  --
  name : VARCHAR(50)
  age : INTEGER "age >= 0 and age <= 200"
  gender : VARCHAR(50)
}

entity Clothing {
  clothing_id : SERIAL
  human_id : INTEGER
  --
  name : VARCHAR(50)
  brand : VARCHAR(50)
  material : VARCHAR(50)
  price : INTEGER "price >= 0"
}

entity Explore_Buildings {
  id : SERIAL
  --
  human_id : INTEGER
  building_id : INTEGER
  operation_code : VARCHAR(50) UNIQUE
}

entity Building {
  building_id : SERIAL
  --
  name : VARCHAR(50)
  year : INTEGER "year >= 0"
  floors_count : INTEGER "floors_count > 0 and floors_count <= 1000"
}

entity Floor {
  floor_id : SERIAL
  building_id : INTEGER
  --
  number : INTEGER "NOT NULL"
  temperature : NUMERIC(4,1) "temperature >= -273.15 and temperature <= 1000"
  rooms_count : INTEGER "rooms_count >= 0 and rooms_count <= 1000"
}

entity Room {
  room_id : SERIAL
  floor_id : INTEGER
  --
  name : VARCHAR(50)
  number : INTEGER "number >= 0"
  temperature : NUMERIC(4,1) "temperature >= -273.15 and temperature <= 1000"
  humidity : INTEGER "humidity >= 0 and humidity <= 100"
}

entity Machinery {
  machinery_id : SERIAL
  room_id : INTEGER
  --
  name : VARCHAR(50)
  company : VARCHAR(50)
  is_active : BOOLEAN
  temperature : NUMERIC(4,1) "temperature >= -273.15 and temperature <= 1000"
}

entity Furniture {
  furniture_id : SERIAL
  room_id : INTEGER
  --
  name : VARCHAR(50)
  material : VARCHAR(50)
  height : FLOAT "height > 0 and height <= 10"
}

entity Plant {
  plant_id : SERIAL
  room_id : INTEGER
  --
  name : VARCHAR(50)
  height : FLOAT "height > 0 and height <= 10"
  alive : BOOLEAN
}
